function drawMountains(W, horizon, pal, rng, layers) { // Back to front: lighter -> darker const shadeSets = currentInverse ? [210, 160, 105, 45] // inverse mode: still separated after inversion+dither : [150, 105, 60, 18]; // normal mode: clear tonal stepping for (let l = layers - 1; l >= 0; l--) { const yBase = horizon + 4; // Back layers sit a bit higher / flatter, front layers taller const depth = l / Math.max(1, layers - 1); const maxH = horizon * ( l === layers - 1 ? rng.range(0.18, 0.28) : l === 0 ? rng.range(0.34, 0.52) : rng.range(0.24, 0.40) ); const nPts = rng.int(16, 28); const ptX = []; const ptY = []; ptX.push(-20); ptY.push(yBase + 10); for (let i = 1; i <= nPts; i++) { const x = (i / (nPts + 1)) * (W + 40) - 20; // Back layers are smoother, front layers more dramatic const jag = 1 - depth * 0.35; const base = rng.range(0.04, 0.14); const spike = rng.next() < (0.22 + jag * 0.18) ? rng.range(0.50, 0.95) : rng.next() < 0.5 ? rng.range(0.22, 0.48) : rng.range(0.02, 0.16); const h = maxH * (base + spike * (1 - base)); ptX.push(x + rng.range(-W / nPts * 0.22, W / nPts * 0.22)); ptY.push(yBase - h); } ptX.push(W + 20); ptY.push(yBase + 10); const shade = shadeSets[Math.min(l, shadeSets.length - 1)]; ctx.beginPath(); ctx.moveTo(ptX[0], ptY[0]); for (let i = 1; i < ptX.length - 1; i++) { const mx = (ptX[i] + ptX[i + 1]) / 2; const my = (ptY[i] + ptY[i + 1]) / 2; ctx.quadraticCurveTo(ptX[i], ptY[i], mx, my); } ctx.lineTo(ptX[ptX.length - 1], ptY[ptY.length - 1]); ctx.lineTo(W + 20, yBase + 10); ctx.lineTo(-20, yBase + 10); ctx.closePath(); ctx.fillStyle = `rgb(${shade}, ${shade}, ${shade})`; ctx.fill(); } }
